async function fetchReview() { const name = document.getElementById("name").value; const url = document.getElementById("url").value; const result = document.getElementById("result"); result.innerHTML = "Searching..."; const res = await fetch("/fetch-review", { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ productUrl: url, reviewerName: name }) }); const data = await res.json(); if (data.success) { result.innerHTML = `${data.permalink}`; } else { result.innerHTML = data.error; } }
